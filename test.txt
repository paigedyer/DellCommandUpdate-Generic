import subprocess

def run_dcu_scan():
    try:
        # Replace 'dcu-cli.exe' with the actual path to the executable if needed
        result = subprocess.run(["dcu-cli.exe", "--scan"], capture_output=True, text=True, check=True)

        # Printing the standard output of the command
        print(result.stdout)

    except subprocess.CalledProcessError as e:
        # Handle any errors that occurred during the execution
        print("Error:", e)
        print("Error Output:", e.stderr)

if __name__ == "__main__":
    run_dcu_scan()
